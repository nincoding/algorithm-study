function solution(maps) {
  let answer = 0;
  let dx = [0, 1, 0, -1];
  let dy = [1, 0, -1, 0];
  const n = maps.length;
  const m = maps[0].length;

    const [x, y, count] = queue.shift();

    for (let i = 0; i < 4; i++) {

}
